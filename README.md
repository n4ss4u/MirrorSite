# Comandos para la instalación
- apt update && apt-get upgrade -y
- apt install git
- git clone https://github.com/n4ss4u/MirrorSite
- apt install python3
- cd MirrorSite
- chmod 777 mirrorsite.py
- pip install -r requirements.txt
- python3 mirrorsite.py
